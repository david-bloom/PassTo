### Full implementation (v4 — deployed 2026-05-28)

Changes from v3: Steps 9–11 replaced with atomic `complete_identity_verification()` RPC (Migration D). Profile update and audit insert now occur in a single Postgres transaction — neither commits if either fails. `writeAuditOrThrow()` removed; `writeAudit()` (swallow) remains for failure paths only. `audit_write_failed` error key no longer emitted; `exchange_failed` idempotency covers the profile-already-advanced case. Two items remain pending sandbox run: IAL field name pin, TEFCA policy claim validation (see Codex Re-QA Remediation sections below).

```typescript
import { serve } from "https://deno.land/std@0.168.0/http/server.ts";
import { createClient } from "https://esm.sh/@supabase/supabase-js@2";

const corsHeaders = {
  "Access-Control-Allow-Origin": "*",
  "Access-Control-Allow-Headers":
    "authorization, x-client-info, apikey, content-type",
};

// RFC 7636 §4.1 — code_verifier: 43–128 unreserved chars
const PKCE_VERIFIER_RE = /^[A-Za-z0-9\-._~]{43,128}$/;

serve(async (req) => {
  if (req.method === "OPTIONS") {
    return new Response("ok", { headers: corsHeaders });
  }

  const supabaseAdmin = createClient(
    Deno.env.get("SUPABASE_URL") ?? "",
    Deno.env.get("SUPABASE_SERVICE_ROLE_KEY") ?? "",
  );

  // ── 1. Parse and validate input ────────────────────────────────────────────
  let code: string;
  let code_verifier: string;

  try {
    const body = await req.json();
    code = body?.code;
    code_verifier = body?.code_verifier;
  } catch {
    return json({ error: "missing_code" }, 400);
  }

  if (!code || typeof code !== "string" || !code.trim()) {
    return json({ error: "missing_code" }, 400);
  }

  // P2.3 fix: enforce PKCE verifier format per RFC 7636 (43–128 unreserved chars)
  if (
    !code_verifier ||
    typeof code_verifier !== "string" ||
    !PKCE_VERIFIER_RE.test(code_verifier)
  ) {
    return json({ error: "invalid_code_verifier" }, 400);
  }

  // ── 2. Verify JWT ───────────────────────────────────────────────────────────
  const authHeader = req.headers.get("Authorization") ?? "";
  if (!authHeader.startsWith("Bearer ")) {
    return json({ error: "unauthenticated" }, 401);
  }
  const jwt = authHeader.slice(7);

  const {
    data: { user },
    error: authError,
  } = await supabaseAdmin.auth.getUser(jwt);

  if (authError || !user) {
    return json({ error: "unauthenticated" }, 401);
  }

  // ── 3. Look up profile ──────────────────────────────────────────────────────
  const { data: profile, error: profileError } = await supabaseAdmin
    .from("profiles")
    .select("id, id_verification_status")
    .eq("auth_user_id", user.id)
    .maybeSingle();

  if (profileError || !profile) {
    return json({ error: "profile_not_found" }, 404);
  }

  // ── 4. Exchange code with ID.me token endpoint ──────────────────────────────
  const tokenParams = new URLSearchParams({
    grant_type: "authorization_code",
    code,
    client_id: Deno.env.get("IDME_CLIENT_ID") ?? "",
    client_secret: Deno.env.get("IDME_CLIENT_SECRET") ?? "",
    redirect_uri: Deno.env.get("IDME_REDIRECT_URI") ?? "",
    code_verifier,
  });

  let accessToken: string;

  try {
    const tokenRes = await fetch(Deno.env.get("IDME_TOKEN_URL") ?? "", {
      method: "POST",
      headers: { "Content-Type": "application/x-www-form-urlencoded" },
      body: tokenParams.toString(),
    });

    if (!tokenRes.ok) {
      const errBody = await tokenRes.text().catch(() => "(unreadable)");
      console.error("ID.me token exchange failed:", tokenRes.status, errBody);
      await writeAudit(supabaseAdmin, profile.id, "identity.verification_failed", {
        error_key: "exchange_failed",
      });
      return json({ verified: false, error: "exchange_failed" }, 200);
    }

    const tokenData = await tokenRes.json();
    accessToken = tokenData?.access_token;

    if (!accessToken) {
      console.error("ID.me token response missing access_token. Keys:", Object.keys(tokenData ?? {}));
      await writeAudit(supabaseAdmin, profile.id, "identity.verification_failed", {
        error_key: "exchange_failed",
      });
      return json({ verified: false, error: "exchange_failed" }, 200);
    }
  } catch (e) {
    console.error("ID.me token exchange threw:", e);
    await writeAudit(supabaseAdmin, profile.id, "identity.verification_failed", {
      error_key: "provider_error",
    });
    return json({ verified: false, error: "provider_error" }, 200);
  }

  // ── 5. Fetch UserInfo from ID.me (OIDC) ────────────────────────────────────
  let attributesData: Record<string, unknown>;

  try {
    const attrRes = await fetch(Deno.env.get("IDME_ATTRIBUTES_URL") ?? "", {
      headers: { Authorization: `Bearer ${accessToken}` },
    });

    if (!attrRes.ok) {
      const errBody = await attrRes.text().catch(() => "(unreadable)");
      console.error("ID.me UserInfo failed:", attrRes.status, errBody);
      await writeAudit(supabaseAdmin, profile.id, "identity.verification_failed", {
        error_key: "provider_error",
      });
      return json({ verified: false, error: "provider_error" }, 200);
    }

    attributesData = await attrRes.json();

    // Diagnostic log — review in Supabase Edge Function logs after sandbox run
    // to confirm exact IAL and subject field names. Remove before production.
    console.log("ID.me UserInfo top-level keys:", Object.keys(attributesData ?? {}));
    console.log("ID.me UserInfo (sanitized):", JSON.stringify({
      sub: attributesData?.sub,
      uuid: attributesData?.uuid,
      ial: attributesData?.ial,
      loa: attributesData?.loa,
      vot: attributesData?.vot,
      vtm: attributesData?.vtm,
      verified: attributesData?.verified,
    }));
  } catch (e) {
    console.error("ID.me UserInfo threw:", e);
    await writeAudit(supabaseAdmin, profile.id, "identity.verification_failed", {
      error_key: "provider_error",
    });
    return json({ verified: false, error: "provider_error" }, 200);
  }

  // ── 6. Validate IAL level ───────────────────────────────────────────────────
  // For OIDC, ID.me returns assurance level in one of:
  //   ial:  numeric (1, 2, 3) — primary check
  //   loa:  NIST URL string, e.g. "http://idmanagement.gov/ns/assurance/ial/2"
  //
  // P2.1 open item: after sandbox run with openid+tefca scope, confirm which
  // field carries the TEFCA assurance claim and add explicit policy validation.
  let ialLevel = 0;
  const ialRaw = attributesData?.ial;
  const loaRaw = attributesData?.loa;

  if (typeof ialRaw === "number") {
    ialLevel = ialRaw;
  } else if (typeof ialRaw === "string") {
    ialLevel = parseInt(ialRaw, 10) || 0;
  } else if (typeof loaRaw === "string") {
    if (loaRaw.includes("ial/2") || loaRaw.includes("ial/3")) ialLevel = 2;
    else if (loaRaw.includes("ial/1")) ialLevel = 1;
  }

  if (ialLevel < 2) {
    console.error("IAL check failed. ial:", ialRaw, "loa:", loaRaw, "resolved ialLevel:", ialLevel);
    await supabaseAdmin
      .from("profiles")
      .update({
        id_verification_status: "failed",
        id_verification_level: "IAL1",
      })
      .eq("id", profile.id);

    await writeAudit(supabaseAdmin, profile.id, "identity.verification_failed", {
      id_verification_status: "failed",
      id_verification_level: "IAL1",
      error_key: "ial1_only",
    });
    return json({ verified: false, error: "ial1_only" }, 200);
  }

  // ── 7. Extract id_me_subject ────────────────────────────────────────────────
  // OIDC standard: 'sub' is the primary, guaranteed subject identifier.
  // 'uuid' is an ID.me-specific fallback if sub is not returned.
  // After sandbox run, confirm which field is the stable non-rotating identifier.
  const subject =
    (attributesData?.sub as string | undefined) ??
    (attributesData?.uuid as string | undefined);

  if (!subject) {
    console.error(
      "ID.me UserInfo missing 'sub' and 'uuid'. UserInfo keys:",
      Object.keys(attributesData ?? {}),
    );
    await writeAudit(supabaseAdmin, profile.id, "identity.verification_failed", {
      error_key: "provider_error",
    });
    return json({ verified: false, error: "provider_error" }, 200);
  }

  // ── 8. Pre-check for duplicate subject (advisory — RPC handles race) ────────
  const { data: existingProfile } = await supabaseAdmin
    .from("profiles")
    .select("id")
    .eq("id_me_subject", subject)
    .neq("id", profile.id)
    .maybeSingle();

  if (existingProfile) {
    await writeAudit(supabaseAdmin, profile.id, "identity.verification_failed", {
      error_key: "identity_already_registered",
    });
    return json({ verified: false, error: "identity_already_registered" }, 200);
  }

  // ── 9. Atomically update profile + write audit (Migration D) ─────────────
  // complete_identity_verification() updates profiles and inserts audit_events
  // in a single Postgres transaction. Neither commits if either fails.
  const { error: rpcError } = await supabaseAdmin.rpc(
    "complete_identity_verification",
    { p_profile_id: profile.id, p_subject: subject },
  );

  if (rpcError) {
    const isUniqueViolation =
      rpcError.code === "23505" ||
      (rpcError.message ?? "").toLowerCase().includes("unique") ||
      (rpcError.message ?? "").toLowerCase().includes("duplicate");

    // Profile already advanced past 'identity' step
    const isWrongStep =
      (rpcError.message ?? "").includes("profile_not_in_identity_step");

    if (isWrongStep) {
      // Return exchange_failed so IdmeCallback idempotency fallback detects
      // the verified profile state and routes the nurse forward.
      await writeAudit(supabaseAdmin, profile.id, "identity.verification_failed", {
        error_key: "exchange_failed",
      });
      return json({ verified: false, error: "exchange_failed" }, 200);
    }

    await writeAudit(supabaseAdmin, profile.id, "identity.verification_failed", {
      error_key: isUniqueViolation ? "identity_already_registered" : "provider_error",
    });
    return json(
      {
        verified: false,
        error: isUniqueViolation ? "identity_already_registered" : "provider_error",
      },
      200,
    );
  }

  // ── 10. Return success ──────────────────────────────────────────────────────
  return json({ verified: true }, 200);
});

// ── Helpers ───────────────────────────────────────────────────────────────────

function json(body: unknown, status: number): Response {
  return new Response(JSON.stringify(body), {
    status,
    headers: { ...corsHeaders, "Content-Type": "application/json" },
  });
}

/** Failure-path audit: swallows errors so the primary error response is not blocked. */
async function writeAudit(
  supabase: ReturnType<typeof createClient>,
  actorId: string,
  action: string,
  changeAfter: Record<string, unknown>,
): Promise<void> {
  try {
    await supabase.from("audit_events").insert({
      actor_id: actorId,
      action,
      resource_type: "profile",
      resource_id: actorId,
      change_after: changeAfter,
    });
  } catch (e) {
    console.error("audit_events write failed (failure path):", e);
  }
}
```

### Migration D — Required Before Sandbox Run (Step 3.1-3a)

David must apply this SQL in Supabase SQL Editor (project `wvzjfxacykgsaffskgtr`) before the sandbox run.

```sql
CREATE OR REPLACE FUNCTION public.complete_identity_verification(
  p_profile_id uuid,
  p_subject    text
)
RETURNS void
LANGUAGE plpgsql
SECURITY DEFINER
SET search_path TO 'public'
AS $function$
begin
  UPDATE public.profiles
  SET
    id_verification_status = 'verified',
    id_verification_level  = 'IAL2',
    id_me_subject          = p_subject,
    onboarding_step        = 'phone',
    updated_at             = now()
  WHERE id = p_profile_id
    AND onboarding_step = 'identity';

  IF NOT FOUND THEN
    RAISE EXCEPTION 'profile_not_in_identity_step: %', p_profile_id;
  END IF;

  INSERT INTO public.audit_events (actor_id, action, resource_type, resource_id, change_after)
  VALUES (
    p_profile_id,
    'identity.verification_completed',
    'profile',
    p_profile_id,
    jsonb_build_object(
      'id_verification_status', 'verified',
      'id_verification_level',  'IAL2',
      'onboarding_step',        'phone'
    )
  );
END;
$function$;
```
