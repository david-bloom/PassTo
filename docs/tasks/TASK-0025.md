## Purpose

Execute the Phase 3.1 implementation defined in TASK-0024. Deploy the `idme-exchange` Supabase Edge Function and update the two Lovable P2 enrollment components (`IdVerification.tsx`, `IdmeCallback.tsx`) to wire the ID.me OAuth flow end-to-end.

After this task:
- A nurse can click "Verify with ID.me" and complete IAL2 identity verification
- The browser handles PKCE/state generation and OAuth redirect only — the token exchange, attribute fetch, and profile write happen server-side
- `profiles.id_verification_status`, `id_verification_level`, `id_me_subject`, and `onboarding_step` are set by the Edge Function
- Successful verification routes the nurse to `/verify-phone` (Phase 3.2)

Phase 3.1 does **not** implement Twilio phone verification, license lookup, data match, selfie confirmation, credential issuance, wallet issuance, Stripe, verifier flows, or dashboard.

### Full implementation

**⚠️ Before deploying:** update the two `TODO` blocks marked below with the confirmed ID.me sandbox response field paths (IAL field, subject field). These depend on Pre-Implementation Gate 2 (protocol) and will be pinned once David confirms the attributes response format.

```typescript
import { serve } from "https://deno.land/std@0.168.0/http/server.ts";
import { createClient } from "https://esm.sh/@supabase/supabase-js@2";

const corsHeaders = {
  "Access-Control-Allow-Origin": "*",
  "Access-Control-Allow-Headers":
    "authorization, x-client-info, apikey, content-type",
};

serve(async (req) => {
  if (req.method === "OPTIONS") {
    return new Response("ok", { headers: corsHeaders });
  }

  const supabaseAdmin = createClient(
    Deno.env.get("SUPABASE_URL") ?? "",
    Deno.env.get("SUPABASE_SERVICE_ROLE_KEY") ?? "",
  );

  // ── 1. Parse and validate input ────────────────────────────────────────────
  let code: string;
  let code_verifier: string;

  try {
    const body = await req.json();
    code = body?.code;
    code_verifier = body?.code_verifier;
  } catch {
    return json({ error: "missing_code" }, 400);
  }

  if (!code || typeof code !== "string" || !code.trim()) {
    return json({ error: "missing_code" }, 400);
  }
  if (
    !code_verifier ||
    typeof code_verifier !== "string" ||
    !code_verifier.trim()
  ) {
    return json({ error: "missing_code_verifier" }, 400);
  }

  // ── 2. Verify JWT ───────────────────────────────────────────────────────────
  const authHeader = req.headers.get("Authorization") ?? "";
  if (!authHeader.startsWith("Bearer ")) {
    return json({ error: "unauthenticated" }, 401);
  }
  const jwt = authHeader.slice(7);

  const {
    data: { user },
    error: authError,
  } = await supabaseAdmin.auth.getUser(jwt);

  if (authError || !user) {
    return json({ error: "unauthenticated" }, 401);
  }

  // ── 3. Look up profile ──────────────────────────────────────────────────────
  const { data: profile, error: profileError } = await supabaseAdmin
    .from("profiles")
    .select("id, id_verification_status")
    .eq("auth_user_id", user.id)
    .maybeSingle();

  if (profileError || !profile) {
    return json({ error: "profile_not_found" }, 404);
  }

  // ── 4. Exchange code with ID.me token endpoint ──────────────────────────────
  const tokenParams = new URLSearchParams({
    grant_type: "authorization_code",
    code,
    client_id: Deno.env.get("IDME_CLIENT_ID") ?? "",
    client_secret: Deno.env.get("IDME_CLIENT_SECRET") ?? "",
    redirect_uri: Deno.env.get("IDME_REDIRECT_URI") ?? "",
    code_verifier,
  });

  let accessToken: string;

  try {
    const tokenRes = await fetch(Deno.env.get("IDME_TOKEN_URL") ?? "", {
      method: "POST",
      headers: { "Content-Type": "application/x-www-form-urlencoded" },
      body: tokenParams.toString(),
    });

    if (!tokenRes.ok) {
      await writeAudit(supabaseAdmin, profile.id, "identity.verification_failed", {
        error_key: "exchange_failed",
      });
      return json({ verified: false, error: "exchange_failed" }, 200);
    }

    const tokenData = await tokenRes.json();
    accessToken = tokenData?.access_token;

    if (!accessToken) {
      await writeAudit(supabaseAdmin, profile.id, "identity.verification_failed", {
        error_key: "exchange_failed",
      });
      return json({ verified: false, error: "exchange_failed" }, 200);
    }
  } catch {
    await writeAudit(supabaseAdmin, profile.id, "identity.verification_failed", {
      error_key: "provider_error",
    });
    return json({ verified: false, error: "provider_error" }, 200);
  }

  // ── 5. Fetch user attributes from ID.me ────────────────────────────────────
  let attributesData: Record<string, unknown>;

  try {
    const attrRes = await fetch(Deno.env.get("IDME_ATTRIBUTES_URL") ?? "", {
      headers: { Authorization: `Bearer ${accessToken}` },
    });

    if (!attrRes.ok) {
      await writeAudit(supabaseAdmin, profile.id, "identity.verification_failed", {
        error_key: "provider_error",
      });
      return json({ verified: false, error: "provider_error" }, 200);
    }

    attributesData = await attrRes.json();
  } catch {
    await writeAudit(supabaseAdmin, profile.id, "identity.verification_failed", {
      error_key: "provider_error",
    });
    return json({ verified: false, error: "provider_error" }, 200);
  }

  // ── 6. Validate IAL level ───────────────────────────────────────────────────
  // TODO: confirm exact IAL field path from ID.me sandbox attributes response.
  // Common paths: attributesData.ial  /  attributesData.attributes?.ial
  // Update before deploying once Pre-Implementation Gate 2 is resolved.
  const ial =
    (attributesData?.ial as number | undefined) ??
    ((attributesData?.attributes as Record<string, unknown>)
      ?.ial as number | undefined) ??
    0;

  if (Number(ial) < 2) {
    await supabaseAdmin
      .from("profiles")
      .update({
        id_verification_status: "failed",
        id_verification_level: "IAL1",
      })
      .eq("id", profile.id);

    await writeAudit(supabaseAdmin, profile.id, "identity.verification_failed", {
      id_verification_status: "failed",
      id_verification_level: "IAL1",
      error_key: "ial1_only",
    });
    return json({ verified: false, error: "ial1_only" }, 200);
  }

  // ── 7. Extract id_me_subject ────────────────────────────────────────────────
  // TODO: confirm exact subject field name from ID.me sandbox attributes response.
  // Common fields: uuid / sub / id / subject
  // Update before deploying once Pre-Implementation Gate 2 is resolved.
  const subject =
    (attributesData?.uuid as string | undefined) ??
    (attributesData?.sub as string | undefined) ??
    (attributesData?.id as string | undefined);

  if (!subject) {
    await writeAudit(supabaseAdmin, profile.id, "identity.verification_failed", {
      error_key: "provider_error",
    });
    return json({ verified: false, error: "provider_error" }, 200);
  }

  // ── 8. Pre-check for duplicate subject (advisory — step 9 handles race) ────
  const { data: existingProfile } = await supabaseAdmin
    .from("profiles")
    .select("id")
    .eq("id_me_subject", subject)
    .neq("id", profile.id)
    .maybeSingle();

  if (existingProfile) {
    await writeAudit(supabaseAdmin, profile.id, "identity.verification_failed", {
      error_key: "identity_already_registered",
    });
    return json({ verified: false, error: "identity_already_registered" }, 200);
  }

  // ── 9. Update profile ───────────────────────────────────────────────────────
  const { error: updateError } = await supabaseAdmin
    .from("profiles")
    .update({
      id_verification_status: "verified",
      id_verification_level: "IAL2",
      id_me_subject: subject,
      onboarding_step: "phone",
    })
    .eq("id", profile.id);

  if (updateError) {
    // Postgres unique-constraint violation: another profile claimed this subject in the race window
    const isUniqueViolation =
      updateError.code === "23505" ||
      (updateError.message ?? "").toLowerCase().includes("unique");

    await writeAudit(supabaseAdmin, profile.id, "identity.verification_failed", {
      error_key: isUniqueViolation ? "identity_already_registered" : "provider_error",
    });
    return json(
      {
        verified: false,
        error: isUniqueViolation ? "identity_already_registered" : "provider_error",
      },
      200,
    );
  }

  // ── 10. Write success audit ─────────────────────────────────────────────────
  await writeAudit(supabaseAdmin, profile.id, "identity.verification_completed", {
    id_verification_status: "verified",
    id_verification_level: "IAL2",
    onboarding_step: "phone",
  });

  // ── 11. Return success ──────────────────────────────────────────────────────
  return json({ verified: true }, 200);
});

// ── Helpers ───────────────────────────────────────────────────────────────────

function json(body: unknown, status: number): Response {
  return new Response(JSON.stringify(body), {
    status,
    headers: { ...corsHeaders, "Content-Type": "application/json" },
  });
}

async function writeAudit(
  supabase: ReturnType<typeof createClient>,
  actorId: string,
  action: string,
  changeAfter: Record<string, unknown>,
): Promise<void> {
  try {
    await supabase.from("audit_events").insert({
      actor_id: actorId,
      action,
      resource_type: "profile",
      resource_id: actorId,
      change_after: changeAfter,
    });
  } catch (e) {
    // Audit write failure must not block the response
    console.error("audit_events write failed:", e);
  }
}
```
